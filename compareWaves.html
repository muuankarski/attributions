vs. Social blame

cbPalette <- c("#E69F00", "#56B4E9", "#999999", "#009E73", "#F0E442", "#0072B2", 
    "#D55E00", "#CC79A7")
dfPlot <- macro[, c(1:4, 13, 14, 16, 17)]
library(ggplot2)
ggplot(data = dfPlot[dfPlot$year == 2006, ], aes(x = wbgi_vae, y = socialBlame, 
    group = cntry, color = group_general)) + geom_point() + geom_text(data = dfPlot[dfPlot$year == 
    2006, ], aes(x = wbgi_vae, y = socialBlame, group = cntry, label = year), 
    vjust = -1, size = 4) + geom_point(data = dfPlot[dfPlot$year == 2010, ], 
    aes(x = wbgi_vae, y = socialBlame, group = cntry, color = group_general)) + 
    geom_text(data = dfPlot[dfPlot$year == 2010, ], aes(x = wbgi_vae, y = socialBlame, 
        group = cntry, label = year), vjust = -1, size = 4) + geom_line(data = dfPlot, 
    aes(x = wbgi_vae, y = socialBlame, group = cntry)) + geom_text(data = dfPlot[dfPlot$year == 
    2010, ], aes(x = wbgi_vae, y = socialBlame, group = cntry, label = cntry), 
    vjust = 1, size = 4) + scale_colour_manual(values = cbPalette) + theme_minimal() + 
    theme(legend.title = element_blank()) + theme(legend.text = element_text(size = 12)) + 
    theme(legend.position = "top")
